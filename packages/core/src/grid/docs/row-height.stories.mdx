Row and column heights defaults to the `--cds-alias-object-interaction-touch-target` value.
You can customize the row height by defining the `--row-height` and `--column-height` property.

<Story id="stories-grid--row-height" />

```html
<cds-grid style="--row-height: 48px; --column-height: 48px">
  <cds-grid-column>Host</cds-grid-column>
  <cds-grid-column>Status</cds-grid-column>
  <cds-grid-column>CPU</cds-grid-column>
  <cds-grid-column>Memory</cds-grid-column>
  <cds-grid-row>
    <cds-grid-cell>vm-host-001</cds-grid-cell>
    <cds-grid-cell>online</cds-grid-cell>
    <cds-grid-cell>5%</cds-grid-cell>
    <cds-grid-cell>10%</cds-grid-cell>
  </cds-grid-row>
  <cds-grid-footer></cds-grid-footer>
</cds-grid>
```
