class DemoDraggableListController extends LitElement {
  @state() private items = Array.from(Array(10).keys()).map(id => ({ id: `${id}` }));

  protected draggableListController = new DraggableListController(this);

  render() {
    return html`
      <section @cdsDraggableChange=${e => this.sortList(e)}>
        ${this.items.map(
          item => html` <cds-card draggable="true">
            <p>${item.id}</p>
            <cds-action-handle aria-label="sort item ${item.id}"></cds-action-handle>
          </cds-card>`
        )}
        <div draggable="false"></div>
      </section>
    `;
  }
}
