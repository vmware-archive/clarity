import { Component } from '@angular/core';
import { FormGroup, FormControl, Validators } from '@angular/forms';

@Component({
  selector: 'app-root',
  template: `
    <form clrForm [formGroup]="model">
      <cds-datalist control-width="shrink">
        <label>Reactive Datalist</label>
        <input placeholder="No label" formControlName="item" />
        <datalist>
          <option *ngFor="let item of items" [value]="item"></option>
        </datalist>
        <cds-control-message status="success" *ngIf="model.valid">
          Everything is valid
        </cds-control-message>
        <cds-control-message *ngIf="model.dirty && !model.invalid && !model.valid">
          Helper text
        </cds-control-message>
        <cds-control-message status="error" *ngIf="model.dirty && model.invalid">
          There was an error
        </cds-control-message>
      </cds-datalist>

      <cds-button [attr.disabled]="model.invalid || null" (click)="submit()">
        Submit
      </cds-button>
    </form>
  `
})
export class AppComponent {
  items = ['Red', 'Green', 'Blue'];

  model = new FormGroup({
    item: new FormControl('', [Validators.required, Validators.minLength(4)])
  });

  submit() {
    console.log('Form submit', this.model);
  }
}
