import { Component } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';

@Component({
  selector: 'app-root',
  template: `
    <form clrForm [formGroup]="model">
      <cds-password control-width="shrink">
        <label>Password</label>
        <input type="password" placeholder="Type a password" formControlName="input" required />
        <cds-control-message *ngIf="model.dirty && !model.invalid && !model.valid">message text</cds-control-message>
        <cds-control-message *ngIf="model.invalid && model.dirty" status="error">error message</cds-control-message>
      </cds-password>
    </form>
  `,
})
export class AppComponent {
  model = new FormGroup({
    input: new FormControl(null),
  });
}
