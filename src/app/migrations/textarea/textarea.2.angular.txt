import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-root',
  template: `
    <form clrForm [formGroup]="model">
      <clr-textarea-container>
        <label>Textarea</label>
        <textarea clrTextarea formControlName="item"></textarea>
        <clr-control-success>
          Everything is valid
        </clr-control-success>
        <clr-control-helper>
          Helper text
        </clr-control-helper>
        <clr-control-error>
          There was an error
        </clr-control-error>
      </clr-textarea-container>
    </form>
  `
})
export class AppComponent {
  model = new FormGroup({
    item: new FormControl('', [Validators.required, Validators.minLength(4)])
  });
}
