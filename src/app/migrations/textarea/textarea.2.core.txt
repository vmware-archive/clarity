import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-root',
  template: `
    <form clrForm [formGroup]="model">
      <cds-textarea control-width="shrink">
        <label>Textarea</label>
        <textarea formControlName="item"></textarea>
        <cds-control-message status="success" *ngIf="model.valid">
          Everything is valid
        </cds-control-message>
        <cds-control-message *ngIf="model.dirty && !model.invalid && !model.valid">
          Helper text
        </cds-control-message>
        <cds-control-message status="error" *ngIf="model.dirty && model.invalid">
          There was an error
        </cds-control-message>
      </cds-textarea>
    </form>
  `,
})
export class AppComponent {
  model = new FormGroup({
    item: new FormControl('', [Validators.required, Validators.minLength(4)])
  });
}
